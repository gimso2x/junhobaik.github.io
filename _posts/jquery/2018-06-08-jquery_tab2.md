JAVASCRIPT
=====

```javascript
$(function() {

	$('.tab_wrap3').waTabMenu();
	$('.tab_wrap4').waTabMenu();

});//$(function()  end

$.fn.waTabMenu = function(){
		this.each(function(){
		var $tab_menu = $(this).find('ul.tab_menu>li');

		$tab_menu.on('click',function(e) {
			e.preventDefault();

			// tab_menu
			$(this).addClass('current').siblings().removeClass('current');
		});
	});
}
```
